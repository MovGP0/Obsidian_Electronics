## Creating and Placing a Power Symbol
A placed power symbol needs a matching `lib_symbols` entry and a schematic-level `symbol` instance. The `lib_id` references the library symbol, and the `Value` property defines the global net name created by the power symbol.
```lisp
(kicad_sch
  (version 20260306)
  (generator "obsidian-electronics")
  (uuid "11111111-1111-1111-1111-111111111111")
  (paper "A4")
  (lib_symbols
    (symbol "power:GND"
      (power global)
      (pin_numbers (hide yes))
      (pin_names (offset 0) (hide yes))
      (property "Reference" "#PWR" (at 0 -6.35 0) (hide yes))
      (property "Value" "GND" (at 0 -3.81 0))
      (symbol "GND_0_1"
        (polyline
          (pts (xy 0 0) (xy 0 -1.27) (xy 1.27 -1.27) (xy 0 -2.54) (xy -1.27 -1.27) (xy 0 -1.27))
          (stroke (width 0) (type default))
          (fill (type none))
        )
      )
      (symbol "GND_1_1"
        (pin power_in line
          (at 0 0 0)
          (length 0)
          (name "" (effects (font (size 1.27 1.27))))
          (number "1" (effects (font (size 1.27 1.27))))
        )
      )
    )
  )
  (symbol
    (lib_id "power:GND")
    (at 50.8 50.8 0)
    (unit 1)
    (body_style 1)
    (exclude_from_sim no)
    (in_bom yes)
    (on_board yes)
    (in_pos_files yes)
    (dnp no)
    (uuid "33333333-3333-3333-3333-333333333333")
    (property "Reference" "#PWR01" (at 50.8 57.15 0) (hide yes))
    (property "Value" "GND" (at 50.8 55.88 0))
    (pin "1" (uuid "22222222-2222-2222-2222-222222222222"))
    (instances
      (project "Example"
        (path "/11111111-1111-1111-1111-111111111111"
          (reference "#PWR01")
          (unit 1)
        )
      )
    )
  )
)
```
